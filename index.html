const { createApp, ref } = Vue; const PROXY = 'https://proxy.api.030101.xyz/'; createApp({ setup() { const keyword = ref(''); const songs = ref([]); const loading = ref(false); const currentSong = ref(null); const currentPlayUrl = ref(''); const errorMsg = ref(''); // 核心：全 HTTPS 稳定线路池 const getSources = (song) => { const encodedName = encodeURIComponent(song.name + ' ' + song.artist); return [ // 线路 1：咪咕音乐 HTTPS 源 (版权歌最稳，包含周杰伦) `https://api.paugram.com/music/?source=migu&name=${encodedName}`, // 线路 2：网易云官方解析 (HTTPS) `https://api.uomg.com/api/rand.music?format=json&name=${encodedName}`, // 线路 3：第三方聚合解析 `https://music.nxinxz.com/kgqq/tx.php?id=${song.mid}&level=standard&type=mp3`, // 线路 4：兜底线路 `https://api.vvhan.com/api/music?id=${song.mid}` ]; }; const searchMusic = async () => { if (!keyword.value.trim()) return; loading.value = true; errorMsg.value = ''; try { // 保持原有的 QQ 音乐搜索逻辑 const postData = { req_1: { method: "DoSearchForQQMusicDesktop", module: "music.search.SearchCgiService", param: { num_per_page: 20, page_num: 1, query: keyword.value, search_type: 0 } } }; const res = await axios.post(PROXY + 'https://u.y.qq.com/cgi-bin/musicu.fcg', postData); const list = res.data.req_1.data.body.song.list || []; songs.value = list.map(s => ({ id: s.id, mid: s.mid, name: s.name, artist: s.singer.map(si => si.name).join('/'), cover: `https://y.gtimg.cn/music/photo_new/T002R300x300M000${s.album.mid}.jpg` })); } catch (e) { errorMsg.value = "搜索接口响应超时"; } finally { loading.value = false; } }; const playSong = (song) => { currentSong.value = song; errorMsg.value = ''; // 默认尝试第一条线路 currentPlayUrl.value = getSources(song)[0]; }; const handleAudioError = () => { // 自动重试逻辑：如果线路1失败，尝试线路2，以此类推 console.error("当前线路失败，正在尝试切换..."); const sources = getSources(currentSong.value); const currentIdx = sources.indexOf(currentPlayUrl.value); if (currentIdx < sources.length - 1) { currentPlayUrl.value = sources[currentIdx + 1]; errorMsg.value = `正在尝试备用线路 ${currentIdx + 2}...`; } else { errorMsg.value = "所有线路均无法播放，请搜索其他版本"; } }; return { keyword, songs, loading, currentSong, currentPlayUrl, errorMsg, searchMusic, playSong, handleAudioError }; } }).mount('#app');
